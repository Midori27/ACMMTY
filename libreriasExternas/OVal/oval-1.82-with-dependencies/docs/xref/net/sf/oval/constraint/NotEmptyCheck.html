13  package net.sf.oval.constraint;

15  import net.sf.oval.ConstraintTarget;
16  import net.sf.oval.Validator;
17  import net.sf.oval.configuration.annotation.AbstractAnnotationCheck;
18  import net.sf.oval.context.OValContext;
19  
20  /**
21   * @author Sebastian Thomschke
22   */
23  public class NotEmptyCheck extends AbstractAnnotationCheck<NotEmpty>
24  {
25  	private static final long serialVersionUID = 1L;
26  
27  	/**
28  	 * {@inheritDoc}
29  	 */
30  	@Override
31  	protected ConstraintTarget[] getAppliesToDefault()
32  	{
33  		return new ConstraintTarget[]{ConstraintTarget.VALUES};
34  	}
35  
36  	/**
37  	 * {@inheritDoc}
38  	 */
39  	public boolean isSatisfied(final Object validatedObject, final Object valueToValidate, final OValContext context,
40  			final Validator validator)
41  	{
42  		if (valueToValidate == null) return true;
43  
44  		return valueToValidate.toString().length() > 0;
45  	}
46  }
